1) CREATE database shop;
2) CREATE user admin; GRANT ALL PRIVILEGES ON database shop TO admin;
3) CREATE user viewer; GRANT SELECT ON all tables in schema public TO viewer;
4) CREATE TABLE category (
category_id serial PRIMARY KEY,
category_name VARCHAR(60) NOT NULL
)
5) INSERT INTO category(category_name)
VALUES ('bread'), ('dairy'), ('fish'), ('meat'), ('vegetable');
6) CREATE TABLE product (
product_id INT GENERATED ALWAYS AS IDENTITY,
product_name VARCHAR(60) NOT NULL,
product_price DECIMAL(10, 2) NOT NULL,
category_id INT,
PRIMARY KEY(product_id),
CONSTRAINT fk_category
    FOREIGN KEY (category_id)
    REFERENCES category (category_id)
)
7) INSERT INTO product(product_name, product_price, category_id)
     VALUES ('long loaf', 30, 1), ('baguette', 35, 1), ('bun', 25, 1),
            ('milk', 30, 2), ('yogurt', 35, 2), ('sour cream', 20, 2),
            ('salmon', 300, 3), ('carp', 100, 3), ('pike', 150, 3),
            ('mutton', 200, 4), ('pork', 100, 4), ('beef', 150, 4),
            ('cucumber', 25, 5), ('tomato', 35, 5), ('truffle', 300, 5)
8) UPDATE product
SET
    product_price = 350
