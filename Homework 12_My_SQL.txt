4)CREATE TABLE category (
category_id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
category_name VARCHAR(60) NOT NULL
)
5)INSERT INTO category(category_name)
VALUES ('bread'), ('dairy'), ('fish'), ('meat'), ('vegetable');
6) CREATE TABLE product (
product_id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
product_name VARCHAR(60) NOT NULL,
product_price DECIMAL(10, 2) UNSIGNED ZEROFILL NOT NULL,
category_id INT(6) UNSIGNED,

FOREIGN KEY (category_id)
    REFERENCES category (category_id)
)
